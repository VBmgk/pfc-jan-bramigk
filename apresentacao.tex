\section{Mapeamento contínuo-discreto}
% Modelagem
% tempo: 3 min
% - Nesce para entender o programa
% - Actions -> Tactics (Skills?)
% - Posse de bola
% - Move Pass Kick
\frame{%
  \frametitle{Mapeamento contínuo-discreto}
  \begin{block}{}
    \centering
    \begin{itemize}
      \item Ação de um time com a bola é uma lista de:
        \begin{itemize}
          \item robô sem a bola:
            \begin{itemize}
              \item $Move(x, y)$
            \end{itemize}
          \item robô com a bola:
            \begin{itemize}
              \item $Move(x, y)$
              \item $Kick(x, y)$
              \item $Pass(r)$
            \end{itemize}
        \end{itemize}
      \item Ação de um time sem a bola é uma lista de:
        \begin{itemize}
          \item $Move(x, y)$
        \end{itemize}
    \end{itemize}
  \end{block}
}
